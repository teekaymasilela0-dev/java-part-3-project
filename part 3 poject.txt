import java.util.ArrayList;
import java.util.Scanner;


// Message Class

class Message {

    private String messageID;
    private int    messageNumber;
    private String recipient;
    private String message;
    private String messageHash;
    private String sender;

    private static int totalMessages = 0;

    // Constructor
    public Message(int messageNumber, String sender, String recipient, String message) {
        this.messageNumber = messageNumber;
        this.sender        = sender;
        this.recipient     = recipient;
        this.message       = message;

        this.messageID   = String.valueOf((long) (Math.random() * 1000000000));
        this.messageHash = createMessageHash();
        totalMessages++;
    }

    public boolean checkMessageID() {
        return messageID.length() <= 10;
    }

    public String checkRecipientCell() {
        if (recipient.matches("^\\+\\d{10,15}$")) {
            return "Cell phone number successfully captured.";
        }
        return "Cell phone number is incorrectly formatted or does not contain international code. Please correct the number and try again.";
    }

    public String checkMessageLength() {
        if (message.length() <= 250) {
            return "Message ready to send.";
        }
        int excess = message.length() - 250;
        return "Message exceeds 250 characters by " + excess + ", please reduce the size.";
    }

    public String createMessageHash() {
        String[] words   = message.trim().split("\\s+");
        String firstWord = words[0].toUpperCase();
        String lastWord  = words[words.length - 1].toUpperCase();
        return messageNumber + ":" + firstWord + lastWord;
    }

    public String sentMessage()  { return "Message successfully sent."; }

    public String printMessage() {
        return "Message Number : " + messageNumber +
                "\nMessage ID   : " + messageID     +
                "\nMessage Hash : " + messageHash   +
                "\nSender       : " + sender        +
                "\nRecipient    : " + recipient     +
                "\nMessage      : " + message;
    }

    public static int  returnTotalMessages() { return totalMessages; }
    public static void resetTotalMessages()  { totalMessages = 0;    }

    public String getMessageID()     { return messageID;     }
    public String getMessageHash()   { return messageHash;   }
    public String getSender()        { return sender;        }
    public String getRecipient()     { return recipient;     }
    public String getMessage()       { return message;       }
    public int    getMessageNumber() { return messageNumber; }
}

    // Preload test data
    public static void preloadTestData(
            String senderName,
            ArrayList<Message> sentMessages,
            ArrayList<Message> disregardedMessages,
            ArrayList<Message> storedMessages,
            ArrayList<String>  sentMessageIDs,
            ArrayList<String>  sentMessageHashes) {

        String[][] testData = {
                { "+27834557896", "Did you get the cake?",                                        "sent"      },
                { "+27838884567", "Where are you? You are late! I have asked you to be on time.", "stored"    },
                { "+27834484567", "Yohoooo, I am at your gate.",                                  "disregard" },
                { "0838884567",   "It is dinner time !",                                          "invalid"   },
                { "+27838884567", "Ok, I am leaving without you.",                                "stored"    }
        };

        int msgNum = 1;
        for (String[] data : testData) {
            String recipient = data[0];
            String msgText   = data[1];
            String flag      = data[2];

            if (flag.equals("invalid")) {
                System.out.println("[Test Data] Message skipped — invalid recipient number: " + recipient);
                continue;
            }

            Message m = new Message(msgNum++, senderName, recipient, msgText);

            switch (flag) {
                case "sent":
                    sentMessages.add(m);
                    sentMessageIDs.add(m.getMessageID());
                    sentMessageHashes.add(m.getMessageHash());
                    break;
                case "stored":
                    storedMessages.add(m);
                    break;
                case "disregard":
                    disregardedMessages.add(m);
                    break;
            }
        }
    }

    // Validation helpers
    public static boolean checkUserName(String username) {
        return username.contains("_") && username.length() <= 5;
    }

    public static boolean checkPasswordComplexity(String password) {
        return password.matches("^(?=.*[A-Z])(?=.*\\d)(?=.*[^a-zA-Z0-9]).{8,}$");
    }

    public static boolean checkCellPhoneNumber(String number) {
        return number.matches("^\\+\\d{10,15}$");
    }

    public static boolean loginUser(String regUsername, String regPassword,
                                    String loginUsername, String loginPassword) {
        return regUsername.equals(loginUsername) && regPassword.equals(loginPassword);
    }
}

// QuickChatTests

class QuickChatTests {

    private static int passed = 0;
    private static int failed = 0;

    //  Entry point called from the menu
    public static void runAll(String senderName) {
        passed = 0;
        failed = 0;

        System.out.println("========================================");
        System.out.println("         RUNNING UNIT TESTS             ");
        System.out.println("========================================");

        // Fresh arrays for every test run so tests are independent
        ArrayList<Message> sent        = new ArrayList<>();
        ArrayList<Message> disregarded = new ArrayList<>();
        ArrayList<Message> stored      = new ArrayList<>();
        ArrayList<String>  ids         = new ArrayList<>();
        ArrayList<String>  hashes      = new ArrayList<>();

        Main.preloadTestData(senderName, sent, disregarded, stored, ids, hashes);

        testSentArrayPopulated(sent);
        testLongestMessage(sent, stored, disregarded);
        testSearchByMessageID(sent, stored, disregarded);
        testSearchByRecipient(sent, stored, disregarded);
        testDeleteByHash(stored);
        testDisplayReport(sent);

        System.out.println("========================================");
        System.out.println("  Results: " + passed + " passed, " + failed + " failed");
        System.out.println("========================================");
    }

    // TEST 1 : Sent Messages array populated
    // Expects "Did you get the cake?" in sentMessages;
    // "It is dinner time !" must NOT be there (invalid recipient).
    private static void testSentArrayPopulated(ArrayList<Message> sentMessages) {
        System.out.println("\n[TEST 1] Sent Messages array correctly populated");

        boolean foundCake   = false;
        boolean foundDinner = false;

        for (Message m : sentMessages) {
            if (m.getMessage().equals("Did you get the cake?"))  foundCake   = true;
            if (m.getMessage().equals("It is dinner time !"))    foundDinner = true;
        }

        assertEqual("sentMessages contains 'Did you get the cake?'",   true,  foundCake);
        assertEqual("'It is dinner time !' excluded (invalid number)",  false, foundDinner);
    }

    // TEST 2 : Display the longest message
    private static void testLongestMessage(ArrayList<Message> sent,
                                           ArrayList<Message> stored,
                                           ArrayList<Message> disregarded) {
        System.out.println("\n[TEST 2] Display the longest message");

        ArrayList<Message> all = new ArrayList<>();
        all.addAll(sent);
        all.addAll(stored);
        all.addAll(disregarded);

        assertNotEmpty("Combined message list is not empty", all);

        Message longest = all.get(0);
        for (Message m : all) {
            if (m.getMessage().length() > longest.getMessage().length()) longest = m;
        }

        assertEqual("Longest message matches expected",
                "Where are you? You are late! I have asked you to be on time.",
                longest.getMessage());
    }

    // TEST 3 : Search for a messageID
    private static void testSearchByMessageID(ArrayList<Message> sent,
                                              ArrayList<Message> stored,
                                              ArrayList<Message> disregarded) {
        System.out.println("\n[TEST 3] Search for a messageID");

        ArrayList<Message> all = new ArrayList<>();
        all.addAll(sent);
        all.addAll(stored);
        all.addAll(disregarded);

        // Positive: look up message 1's real ID
        String knownID   = sent.get(0).getMessageID();
        String foundBody = "";
        for (Message m : all) {
            if (m.getMessageID().equals(knownID)) { foundBody = m.getMessage(); break; }
        }
        assertEqual("Search by valid ID returns correct message body",
                "Did you get the cake?", foundBody);

        // Negative: ID that doesn't exist
        boolean notFound = true;
        for (Message m : all) {
            if (m.getMessageID().equals("0000000000")) { notFound = false; break; }
        }
        assertEqual("Search for non-existent ID returns no result", true, notFound);
    }

    // TEST 4 : Search all messages for a particular recipient
    private static void testSearchByRecipient(ArrayList<Message> sent,
                                              ArrayList<Message> stored,
                                              ArrayList<Message> disregarded) {
        System.out.println("\n[TEST 4] Search all messages for a particular recipient");

        String target = "+27838884567";
        ArrayList<Message> all = new ArrayList<>();
        all.addAll(sent);
        all.addAll(stored);
        all.addAll(disregarded);

        ArrayList<String> found = new ArrayList<>();
        for (Message m : all) {
            if (m.getRecipient().equals(target)) found.add(m.getMessage());
        }

        assertEqual("Exactly 2 messages found for +27838884567", 2, found.size());
        assertEqual("First stored message present",
                true,
                found.contains("Where are you? You are late! I have asked you to be on time."));
        assertEqual("Second stored message present",
                true,
                found.contains("Ok, I am leaving without you."));
    }

    // TEST 5 : Delete a message using a message hash
    private static void testDeleteByHash(ArrayList<Message> storedMessages) {
        System.out.println("\n[TEST 5] Delete a message using a message hash");

        ArrayList<Message> copy = new ArrayList<>(storedMessages);

        assertNotEmpty("storedMessages is not empty before deletion", copy);

        // Locate Test Message 2 by its content
        Message target = null;
        for (Message m : copy) {
            if (m.getMessage().equals(
                    "Where are you? You are late! I have asked you to be on time.")) {
                target = m;
                break;
            }
        }

        assertNotNull("Test Message 2 found in storedMessages", target);

        int    sizeBefore    = copy.size();
        String hashToDelete  = target.getMessageHash();
        String confirmation  = "";
        boolean deleted      = false;

        for (int i = 0; i < copy.size(); i++) {
            if (copy.get(i).getMessageHash().equals(hashToDelete)) {
                Message removed = copy.remove(i);
                confirmation = "Message: \"" + removed.getMessage() + "\" successfully deleted.";
                deleted = true;
                break;
            }
        }

        assertEqual("Message was deleted",                  true, deleted);
        assertEqual("List shrinks by 1 after deletion",     sizeBefore - 1, copy.size());
        assertEqual("Confirmation contains deleted message text",
                true,
                confirmation.contains(
                        "Where are you? You are late! I have asked you to be on time."));

        System.out.println("  Confirmation: " + confirmation);
    }

    // TEST 6 : Display Report
    private static void testDisplayReport(ArrayList<Message> sentMessages) {
        System.out.println("\n[TEST 6] Display Report");

        assertNotEmpty("sentMessages is not empty for report", sentMessages);

        StringBuilder report = new StringBuilder();
        report.append("=== Full Report: Sent Messages ===\n");
        report.append("Total sent: ").append(sentMessages.size()).append("\n");

        for (Message m : sentMessages) {
            report.append("----------------------------------\n");
            report.append("Message Hash : ").append(m.getMessageHash()).append("\n");
            report.append("Recipient    : ").append(m.getRecipient()).append("\n");
            report.append("Message      : ").append(m.getMessage()).append("\n");
        }

        String r = report.toString();

        assertEqual("Report contains 'Message Hash' heading",  true, r.contains("Message Hash"));
        assertEqual("Report contains 'Recipient' heading",     true, r.contains("Recipient"));
        assertEqual("Report contains 'Message' heading",       true, r.contains("Message"));
        assertEqual("Report contains message 1 recipient",     true, r.contains("+27834557896"));
        assertEqual("Report contains message 1 body",          true, r.contains("Did you get the cake?"));
        assertEqual("Report contains message 1 hash",          true, r.contains(sentMessages.get(0).getMessageHash()));

        System.out.println("  --- Report Preview ---");
        System.out.print(r);
    }

    // Assertion helpers
    private static void assertEqual(String description, Object expected, Object actual) {
        if (expected.equals(actual)) {
            System.out.println("  PASS : " + description);
            passed++;
        } else {
            System.out.println("  FAIL : " + description);
            System.out.println("         Expected : " + expected);
            System.out.println("         Actual   : " + actual);
            failed++;
        }
    }

    private static void assertNotEmpty(String description, ArrayList<?> list) {
        if (!list.isEmpty()) {
            System.out.println("  PASS : " + description);
            passed++;
        } else {
            System.out.println("  FAIL : " + description + " (list was empty)");
            failed++;
        }
    }

    private static void assertNotNull(String description, Object obj) {
        if (obj != null) {
            System.out.println("  PASS : " + description);
            passed++;
        } else {
            System.out.println("  FAIL : " + description + " (was null)");
            failed++;
        }
    }
